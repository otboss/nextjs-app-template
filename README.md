<h1>NextJS App Template</h1>
<h2>A quickstart next js app</h2>
<ol>
    <div>
    <li>Install dependencies:</li>
    <pre>npm install -g typescript; npm install;</pre>
    </div>
    <div>
    <li>Transpile ts files</li>
    <pre>tsc src/index.ts</pre>
    </div>
    <div>
    <li>Run server</li>
    <pre>npm run debug</pre>
    </div>
</ol>
